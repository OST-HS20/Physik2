\newenvironment{formula}[1]{
	\noindent\par 
	\begin{tabular*}{\columnwidth}{l  @{\extracolsep{\fill}} r}
		\colorboxed{red}{
			\begin{aligned}#1\end{aligned}
		}
		&
		\begin{tabular}{
				>{$}l<{$}
				@{${}:{}$} l
				>{$}r<{$}
			}
} {
		\end{tabular}
	\end{tabular*}
}

}
